const orderInfo = {
  appId: 'wx1234567890abcdef',
  timeStamp: '1516132534',
  nonceStr: '5K8264ILTKCH16CQ2502SI8ZNMTM67VS',
  package: 'prepay_id=wx2017033010242291fcfe0db70013231072',
  signType: 'MD5'

 const apiKey = 'abcdefghijklmnopqrstuvwxyz123456';
 // 计算签名
const paySign = createPaySign(orderInfo, apiKey);
 // 根据API密钥和订单信息生成签名
function createPaySign(orderInfo, apiKey) {
  const arr = [];
  for (const key in orderInfo) {
    if (orderInfo.hasOwnProperty(key) && orderInfo[key]) {
      arr.push(key + '=' + orderInfo[key]);
    }
  }
  arr.sort();
  arr.push('key=' + apiKey);
  const str = arr.join('&');
  return md5(str).toUpperCase();
}
